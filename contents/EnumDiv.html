ある正整数nが与えられたときに、約数を全列挙するプログラム

計算量はO(\sqrt a)


#include <algorithm>
#include <vector>

vector<long long> enum_div(const long long& n) {
    vector<long long> ans;
    for (long long i=1;i*i<=n;++i) {
        if (n % i == 0) {
            ans.push_back(i);
            if (i != n/i) ans.push_back(n/i);
        }
    }
    sort(ans.begin(), ans.end());
    return ans;
}
