左上から平面操作をする際の順番を返すプログラム


    vector<pair<int,int>> oblique_scanning(const int& h, const int& w) {
  /**
   * | 1| 2| 4| 7|
   * | 3| 5| 8|10|
   * | 6| 9|11|12|
  */
  vector<pair<int,int>> ans;
  for (int i=0;i<h+w-1;++i) {
    int s = min(w-1, i);
    int f = i - s;
    for (;f < h && 0 <= s;++f, --s) {
      ans.push_back(make_pair(f,s));
    }
  }
  return ans;
}
